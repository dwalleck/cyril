function resolveRemoteToolAllowlist(client2, channel, options2) {
  if (process.env[KIRO_LOAD_ALL_REMOTE_TOOLS_ENV] === "true") return "*";
  const memoryEnabled = options2?.memoryEnabled ?? false;
  switch (client2) {
    case "kiro-web":
      return "*";
    case "kiro-ide": {
      if (channel === "stable") return [REMOTE_TOOL_WEB_SEARCH];
      if (channel === "insider") return [REMOTE_TOOL_WEB_SEARCH, REMOTE_TOOL_SEARCH_MEMORIES];
      return [REMOTE_TOOL_WEB_SEARCH];
    }
    case "kiro-cli": {
      const tools = [REMOTE_TOOL_WEB_SEARCH];

    default:
      return [REMOTE_TOOL_WEB_SEARCH];
  }
}
